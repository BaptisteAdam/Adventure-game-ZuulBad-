001package pkg_Command;
002import pkg_Engine.GameEngine;
003
004import java.util.Scanner;
005import java.io.FileNotFoundException;

007/**
008 * La commande permettant de prendre un objet.
009 * 
010 * @author Michael Kolling and David J. Barnes
011 * @version 2011.07.31
012 */
013public class TakeCommand extends Command
014{
015    private CommandWords commandWords;
016    
017    /**
018     * Constructeur, cree une nouvelle TakeCommand.
019     * @param words Le CommandWords correspondant
020     */
021    public TakeCommand(CommandWords words)
022    {
023        commandWords = words;
024    }
025    
026    /**
027     * Execute la commande Take.
028     * @param pGameEngine Le GameEngine faisant l'action
029     */
030    public void execute(GameEngine pGameEngine)
031    {
032        String vItem = getSecondWord();
033        if(vItem == null)
034            pGameEngine.getGUI().println("Take what ? \n");
035        else
036        {
037            String vS;
038            if(vItem.equals("orb"))
039                vS = pGameEngine.getPlayer().takeBeamer();
040            else
041                vS = pGameEngine.getPlayer().takeTest(vItem);
042            pGameEngine.getGUI().println(vS);
043            if(vS.equals("You have retrieved the statuette of Anbrodema."))
044                   pGameEngine.endGameWin(); 
045        }
046    }
047}
