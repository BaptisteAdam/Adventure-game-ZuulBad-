001package pkg_Command;
002
003import java.util.StringTokenizer;
004import java.io.BufferedReader;
005import java.io.InputStreamReader;
006
007/**
008 * This class is part of the "World of Zuul" application. 
009 * "World of Zuul" is a very simple, text based adventure game.  
010 * 
011 * This parser reads user input and tries to interpret it as an "Adventure"
012 * command. Every time it is called it reads a line from the terminal and
013 * tries to interpret the line as a two word command. It returns the command
014 * as an object of class Command.
015 *
016 * The parser has a set of known command words. It checks user input against
017 * the known commands, and if the input is not one of the known commands, it
018 * returns a command object that is marked as an unknown command.
019 * 
020 * @author  Michael Kolling and David J. Barnes + D.Bureau
021 * @version 2008.03.30 + 2013.09.15
022 */
023public class Parser 
024{
025    private CommandWords aCommandWords;  // (voir la classe CommandWords)
026
027    /**
028     * Constructeur par defaut qui cree les 2 objets prevus pour les attributs
029     */
030    public Parser() 
031    {
032        this.aCommandWords = new CommandWords();
033        // System.in designe le clavier, comme System.out designe l'ecran
034    } // Parser()
035
036    /**
037     * Get a new command from the user. The command is read by
038     * parsing the 'inputLine'.
039     * @param pInputLine Une string qui a ete soit tapee soit recuperee sur un bouton.
040     * @return La commande correspondant a la String en parametre.
041     */
042    public Command getCommand( final String pInputLine ) 
043    {
044        String vWord1;
045        String vWord2;
046
047        StringTokenizer tokenizer = new StringTokenizer( pInputLine );
048
049        if ( tokenizer.hasMoreTokens() )
050            vWord1 = tokenizer.nextToken();      // get first word
051        else
052            vWord1 = null;
053
054        if ( tokenizer.hasMoreTokens() )
055            vWord2 = tokenizer.nextToken();      // get second word
056        else
057            vWord2 = null;
058
059        // note: we just ignore the rest of the input line.
060
061        // Now check whether this word is known. If so, create a command
062        // with it. If not, create a "null" command (for unknown command).
063
064        Command command = aCommandWords.getCommandWord(vWord1);
065        if(command != null) {
066            command.setSecondWord(vWord2);
067        }
068        return command;
069    } // getCommand(.)
070    
071    /**
072     * Affiche toute les commandes valides.
073     * @return La string avec toutes les commandes valides.
074     */
075    public static String showCommands()
076    {
077        return CommandWords.getCommandList();
078    }
079} // Parser
