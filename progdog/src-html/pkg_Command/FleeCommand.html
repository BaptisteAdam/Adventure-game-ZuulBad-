001package pkg_Command;
002import pkg_Engine.GameEngine;
003/**
004 * La commande permettant de se suicider.
005 * 
006 * @author Michael Kolling and David J. Barnes
007 * @version 2011.07.31
008 */
009public class FleeCommand extends Command
010{
011    private CommandWords commandWords;
012    
013    /**
014     * Constructeur, cree une nouvelle FleeCommand.
015     * @param words Le CommandWords correspondant
016     */
017    public FleeCommand(CommandWords words)
018    {
019        commandWords = words;
020    }
021    
022    /**
023     * Execute la commande Flee.
024     * @param pGameEngine Le GameEngine faisant l'action
025     */
026    public void execute(GameEngine pGameEngine)
027    {
028        if(pGameEngine.getPlayer().getAgi()<=0)
029            pGameEngine.death("You were attacked by behind while fleeing and died.");
030        else
031        {
032            pGameEngine.getGUI().panelDeplacement();
033            pGameEngine.getPlayer().changeAgi(-3);
034            pGameEngine.setCombat(false);
035            if(pGameEngine.getPlayer().getImage() != null)
036                pGameEngine.getGUI().showImage(pGameEngine.getPlayer().getImage());
037            String vEnnemi;
038            if(pGameEngine.getPlayer().getMovingCharacter()!=null)
039                vEnnemi = pGameEngine.getPlayer().getMovingCharacter().getNom();
040            else
041                vEnnemi = pGameEngine.getEnnemi().getNom();
042            pGameEngine.getGUI().println("You fled, the " + vEnnemi + " does not see you any more.\n");
043        }
044    }
045}
