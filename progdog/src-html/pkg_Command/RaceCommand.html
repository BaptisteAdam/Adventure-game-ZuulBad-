001package pkg_Command;
002import pkg_Engine.GameEngine;
003/**
004 * La commande permettant de choisir sa Race.
005 * 
006 * @author Michael Kolling and David J. Barnes
007 * @version 2011.07.31
008 */
009public class RaceCommand extends Command
010{
011    private CommandWords commandWords;
012    
013    /**
014     * Constructeur, cree une nouvelle RaceCommand.
015     * @param words Le CommandWords correspondant
016     */
017    public RaceCommand(CommandWords words)
018    {
019        commandWords = words;
020    }
021    
022    /**
023     * Execute la commande de choix de la Race.
024     * @param pGameEngine Le GameEngine faisant l'action
025     */
026    public void execute(GameEngine pGameEngine)
027    {
028        String vRace = getSecondWord();
029        pGameEngine.getPlayer().setRace(vRace);
030        pGameEngine.getGUI().panelClasse();
031        pGameEngine.getGUI().showImage("Images/classe.png");
032        pGameEngine.getGUI().println("Please, choose a job.");
033    }
034}
