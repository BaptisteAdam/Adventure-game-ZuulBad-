001package pkg_Componnants;
002
003import java.util.ArrayList;
004
005/**
006 * Permet de selectionner une Room aleatoirement dans une ArrayList de Room.
007 *
008 * @author Baptiste ADAM
009 * @version v2 - 30/05/2017
010 */
011public class RoomRandomizer
012{
013   private ArrayList<Room> aAllRooms;
014   private Integer aIndiceRoom = null;
015   
016   /**
017    * Constructeur, cree un nouveau RoomRandomizer.
018    * @param pAllRooms Une ArrayList de Room
019    */
020   public RoomRandomizer( final ArrayList<Room> pAllRooms )
021   {
022       this.aAllRooms = pAllRooms;
023   }
024   
025   /**
026    * Permet de set IndiceRoom sur null ou 0.
027    * @param pB Un booleen : true met IndiceRoom a 0 (desactive le random)
028    */
029   public void setIndiceRoom(final boolean pB)
030   {
031       if(pB)
032           aIndiceRoom = 0;
033       else
034           aIndiceRoom = null;
035   }
036   
037   /**
038    * Renvois une Room aleatoirement.
039    * @return Une Room de l'ArrayList
040    */
041   public Room getRandomRoom()
042   {
043       if(aIndiceRoom == null)
044       {
045           int aNbRand = (int)Math.floor(Math.random()*aAllRooms.size());
046           return aAllRooms.get(aNbRand);
047       }
048        else
049           return aAllRooms.get(aIndiceRoom);
050   }
051    
052}
