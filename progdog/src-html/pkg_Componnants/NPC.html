001package pkg_Componnants;

003
004/**
005 * Write a description of class Character here.
006 *
007 * @author (your name)
008 * @version (a version number or a date)
009 */
010public class NPC
011{
012    private String aNom;
013    private String aDialogue;
014    private String aImage;
015    
016    private Statistiques aStats;
017    private int aPV ;
018    private int aDef;
019    private int aAgi;
020    private int aAtt;
021    
022    /**
023     * Constructeur, cree un nouveau NPC.
024     * @param pNom Une String, Le nom du NPC
025     * @param pImage Une String, L'image du NPC
026     * @param pDialogue Un tableau de String, Les phrases que prononce le NPC
027     * @param pRace Une String, La race du NPC
028     * @param pJob Une String, La classe du NPC
029     */
030    public NPC(final String pNom, final String pImage, 
031                     final String[] pDialogue, 
032                     final String pRace, final String pJob)
033    {
034        this.aNom = pNom;
035        this.aImage = pImage;
036        int aNbRand = (int)Math.floor(Math.random()*pDialogue.length);
037        String vDialogue = pDialogue[aNbRand];
038        this.aDialogue = vDialogue;
039        this.aStats = new Statistiques();
040        this.attribuerStats(pRace, pJob);
041    }
042    
043    /**
044    * Attribut les statistiques du NPC grace a sa race et sa classe (job)
045    * @param pRace La arce du NPC 
046    * @param pJob La classe du NPC
047    */
048    public void attribuerStats(final String pRace, final String pJob)
049    {
050       int vPV = aStats.getPV(pRace) + aStats.getPV(pJob);
051       int vDef = aStats.getDef(pRace) + aStats.getDef(pJob);
052       int vAgi = aStats.getAgi(pRace) + aStats.getAgi(pJob);
053       int vAtt = aStats.getAtt(pRace) + aStats.getAtt(pJob);
054       
055       this.aPV = vPV;
056       this.aDef = vDef;
057       this.aAgi = vAgi;
058       this.aAtt = vAtt;
059    }
060    
061    /**
062     * retourne le nom du NPC.
063     * @return Une String
064     */
065    public String getNom()
066    {
067       return this.aNom;
068    }
069   
070    /**
071     * Change le nom du NPC avec le nom passe en parametre.
072     * @param pNom Une String, le nouveau nom du NPC
073     */
074    public void setNom(final String pNom)
075    {
076        this.aNom = pNom;
077    }
078    
079    /**
080     * retourne le dialogue qui a ete selectionne pour le NPC.
081     * @return Une String
082     */
083    public String getDialogue()
084    {
085       return this.aDialogue;
086    }
087    
088    /**
089     * retourne l'image du NPC.
090     * @return Une String
091     */
092    public String getImage()
093    {
094        return this.aImage;
095    }
096    
097    /**
098     * Change l'image du NPC
099     * @param pImage Une String, la nouvelle image du NPC
100     */
101    public void setImage(final String pImage)
102    {
103        this.aImage = pImage;
104    }
105    
106    /**
107    * Retourne les PV actuel du NPC.
108    * @return Un int, les PV du NPC
109    */
110   public int getPV()
111   {
112       return this.aPV;
113   }
114   
115   /**
116    * Permet de faire varier les PV.
117    * @param pPV La valeur a ajouter
118    */
119   public void changePV(final int pPV)
120   {
121       this.aPV += pPV;
122   }
123   
124   /**
125    * Retourne la Defense actuelle du NPC.
126    * @return Un int, la Def du NPC
127    */
128   public int getDef()
129   {
130       return this.aDef;
131   }
132   
133   /**
134    * Permet de faire varier la Def.
135    * @param pDef La valeur a ajouter
136    */
137   public void changeDef(final int pDef)
138   {
139       this.aDef += pDef;
140   }
141   
142   /**
143    * Retourne l'Agilite actuelle du NPC.
144    * @return Un int, l'Agi du NPC
145    */
146   public int getAgi()
147   {
148       return this.aAgi;
149   }
150   
151   /**
152    * Permet de faire varier l'Agi.
153    * @param pAgi La valeur a ajouter
154    */
155   public void changeAgi(final int pAgi)
156   {
157       this.aAgi += pAgi;
158   }
159   
160   /**
161    * Retourne l'Attaque actuelle du NPC.
162    * @return Un int, l'Att du NPC
163    */
164   public int getAtt()
165   {
166       return this.aAtt;
167   }
168   
169   /**
170    * Permet de faire varier l'Att.
171    * @param pAtt La valeur a ajouter
172    */
173   public void changeAtt(final int pAtt)
174   {
175       this.aAtt += pAtt;
176   }
177   
178   /**
179    * Affiche les statistiques du NPC.
180    * @return Une String contenant toutes les statistiques du NPC
181    */
182   public String stats()
183   {
184       String vS = "The " + aNom + " : " +
185                   "\n    PV: " + aPV + "   -   Def: " + aDef 
186                 + "\n    Agi:" + aAgi + "   -   Att: " + aAtt + "\n";
187       return vS;
188   }
189}
