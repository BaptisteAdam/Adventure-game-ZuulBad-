004/**
005 * Cette classe crée une Map pou le jeu. 
006 * Celle-ci est faite aleatoirement avec un nombre 
007 * maximum de Room limitant la taille. Une type est attribué a chaque Room, 
008 * il represente les differentes actions qu'il s'y passera.
009 * De plus, il est assuré qu'une Room de type 5 est ete cree puisque c'est 
010 * celle qui comporte la Statuette, est donc, le seul moyen de gagner.
011 *
012 * @author Baptiste Adam
013 * @version v1 - 13/05/2017
014 */
015public class Map
016{
017   private static int numRoom = 1;
018   private static final double MAX_ROOM = 25; //nombre minimum 5 
019   
020   private ArrayList<Room> aAllRooms ;
021   private TransporterRoom aTransporterRoom;

023   /**
024    * Constructeur, cree une nouvelle Map
025    */
026   public Map()
027   {
028       aAllRooms = new ArrayList<Room>();
029       this.createMap();
030   }

032   /**
033    * S'assure que l'ArrayList est bien vide avant d'y ajouter les nouvelles Rooms.
034    * De plus, recommence l'action autant de fois que necessaire pour qu'il y ais une Room de Type 5,
035    * c'est a dire, pour que le jeu sois finissable.
036    */
037   private void createMap()
038   {
039       int nbType5 = 0;
040       int nbType9 = 0;
041       while(nbType5 != 1 && nbType9 != 1)
042       {
043           nbType5 = 0;
044           nbType9 = 0;
045           this.clearRooms();
046           this.createRooms();
047           for(Room vRoom : aAllRooms)
048           {
049              if (vRoom.getType() == 5)
050                 nbType5++;
051              else if (vRoom.getType() == 9)
052                 nbType9++;
053           }
054       }
055   }
056   
057   /**
058    * Efface toutes les Rooms de l'ArrayList
059    * Reinitialise le Compteur de Room.
060    */
061   private void clearRooms()
062   {
063       this.aAllRooms.clear();
064       numRoom = 1;
065   }

067   /**
068    * Initialise la premiere Room (la porte du donjon) et sa sortie vers la premiere Room vers le nord.
069    * Puis, genere aleatoirement toutes les autres salles}.
070    * S'assure que les sorties ajoutees a chaque salle n'efface pas le chemin permettant de retourner dans la salle precedente.
071    */
072   private void createRooms()
073   {
074       //génère l'entrée du donjon
075       Room vOutside = new Room( "in front of a massive wooden door made of dark ebony wood", "Images/Wooden door.jpg", 0);
076       aAllRooms.add( vOutside );
077       
078       Item vBranch = new Item("You see a tree branch on the ground.", 0.6, "branch");
079       aAllRooms.get(0).addItem(vBranch);
080       
081       //génère la premiere salle du donjon, elle se trouve forcément au nord de l'entrée.
082       aAllRooms.add( this.RoomType6("south", vOutside) );
083       vOutside.setExit("north", aAllRooms.get(1));
084
085       //génère le reste des salles aléatoirement
086       String vDirectionPrecedente = "south"; //la seule sortie de la derniere salle déjà créée. Ne doit pas sercir pour une autre sortie.
087       for(int i=1 ; i<Math.floor(MAX_ROOM/3) ; i++)
088       {
089           //Générer une direction pour sortir de la salle actuelle ( aAllRooms.get(i) )
090           String vDirection = this.genererDirection();
091           while(vDirection.equals(vDirectionPrecedente))
092                vDirection = this.genererDirection();
093           //générer une Room qui permet de traverser dans le sens inverse et de revenir dans la salle actuelle.
094           Room vRoom = this.genererRoom(this.invDirection(vDirection), aAllRooms.get(i) );
095           if(aAllRooms.get(i).getType()!=5)
096               aAllRooms.get(i).setExit(vDirection, vRoom); 
097           //mémoriser la sortie "retour sur ses pas" de la nouvelle salle pour eviter de mettre une autre sortie au même endroit.
098           vDirectionPrecedente = this.invDirection(vDirection);
099           aAllRooms.add(vRoom);
100       }
101       
102       for (double z = Math.floor(MAX_ROOM/3) ; z < MAX_ROOM-2 ; z++)
103       {
104           int aNbRand = (int)Math.floor(Math.random()*(z-1)+1);
105           this.createRoomRamification(aNbRand);
106       }
107       
108       int aNbRand = (int)Math.floor(Math.random()*(MAX_ROOM-2)+1);
109       this.createTransporterRoom(aNbRand);
110    }

112   /**
113    * Cree des ramifications de Rooms sur les Rooms deja existantes.
114    * Si la Room sur laquelle la ramification veut etre cree est de type 5, le raccord n'est pas fait..
115    * (la type 5 étant la fin, il est inutile d'y avoir un acces vers une nouvelle Room)
116    * @param pIndice L'indice de la Room sur laquelle on cree une ramification.
117    */
118    private void createRoomRamification(final int pIndice)
119   {
120       String[] vTabDirection = {"north", "south", "east", "west", "up", "down"};
121       ArrayList<String> vTabDirectionRoom = new ArrayList<String>();
122       for(int i=0 ; i<vTabDirection.length ;i++)
123       {
124           if (aAllRooms.get(pIndice).getExit(vTabDirection[i])!=null)
125               vTabDirectionRoom.add(vTabDirection[i]);
126       }
127       
128       String vDirection = this.genererDirection();;
129       boolean ok = false;
130       while(ok == false)
131       {
132           ok = true;
133           vDirection = this.genererDirection();
134           for(String vExits : vTabDirectionRoom)
135               if(vExits.equals(vDirection) )
136                  ok = false;
137       }
138       
139       Room vRoom2 = genererRoom(this.invDirection(vDirection), aAllRooms.get(pIndice) );
140       if(aAllRooms.get(pIndice).getType()!=5 )
141           aAllRooms.get(pIndice).setExit(vDirection, vRoom2); 
142       aAllRooms.add( vRoom2);
143   }

145   /**
146    * Cree une TransporterRoom sur les Rooms deja existantes.
147    * Si la Room sur laquelle la TransporterRoom veut etre cree est de type 5, le raccord n'est pas fait..
148    * (la type 5 étant la fin, il est inutile d'y avoir un acces vers une nouvelle Room)
149    * @param pIndice L'indice de la Room sur laquelle on cree la TransporterRoom.
150    */
151   public void createTransporterRoom(final int pIndice)
152   {
153       String[] vTabDirection = {"north", "south", "east", "west", "up", "down"};
154       ArrayList<String> vTabDirectionRoom = new ArrayList<String>();
155       for(int i=0 ; i<vTabDirection.length ;i++)
156       {
157           if (aAllRooms.get(pIndice).getExit(vTabDirection[i])!=null)
158               vTabDirectionRoom.add(vTabDirection[i]);
159       }
160       
161       String vDirection = this.genererDirection();;
162       boolean ok = false;
163       while(ok == false)
164       {
165           ok = true;
166           vDirection = this.genererDirection();
167           for(String vExits : vTabDirectionRoom)
168               if(vExits.equals(vDirection) )
169                  ok = false;
170       }
171       
172       aTransporterRoom = RoomType8(this.invDirection(vDirection), aAllRooms.get(pIndice) );
173       if(aAllRooms.get(pIndice).getType()!=5 )
174           aAllRooms.get(pIndice).setExit(vDirection, aTransporterRoom); 
175   }

223   /**
224    * Retourne la direction opposee a celle passee en parametre.
225    * @param pDirection Une direction.
226    * @return La direction opposee.
227    */
228   private String invDirection( final String pDirection)
229   {
230       if(pDirection.equals("north"))
231           return "south";
232       else if(pDirection.equals("south"))
233           return "north";
234       else if(pDirection.equals("west"))
235           return "east";
236       else if(pDirection.equals("east"))
237           return "west";
238       else if(pDirection.equals("up"))
239           return "down";
240       else
241           return "up";
242   }

244   /**
245    * genere une direction aleatoirement.
246    * @return Une direction
247    */
248   public String genererDirection()
249   {
250       double aNbRand = Math.round(Math.random()*6);
251       if(aNbRand == 1.0)
252           return "north";
253       else if(aNbRand == 2.0)
254           return "south";
255       else if(aNbRand == 3.0)
256           return "west";
257       else if(aNbRand == 4.0)
258           return "east";
259       else if(aNbRand == 5.0)
260           return "down";
261       else 
262           return "up";
263   }

293       else if (39.0<=aNbRand && aNbRand<46.0)//fleche dans le dos

296       else if (46.0<=aNbRand && aNbRand<53.0)//trappe
297           return RoomType15(pStringRoomPrecedent, pRoomPrecedente);
298       
299       else if (53.0<=aNbRand && aNbRand<60.0)//transformation poulet
300           return RoomType19(pStringRoomPrecedent, pRoomPrecedente);
301       
302       else if (60.0<=aNbRand && aNbRand<65.0)//magiciens
