001package pkg_Componnants;
002
003import java.util.ArrayList;
004
005/**
006 * TransporterRoom est une Room qui teleporte le Player aleatoirement sur la map
007 * lorsqu'on essaye d'en sortir.
008 *
009 * @author Baptiste ADAM
010 * @version v2 - 30/05/2017
011 */
012public class TransporterRoom extends Room
013{
014   private RoomRandomizer aRoomRandomizer;
015    
016   /**
017    * Constructeur, cree une nouvelle TransporterRoom
018    * @param pDescription La description de la TransporterRoom
019    * @param pImage L'adresse de l'image de la TransporterRoom
020    * @param pTypeRoom Le type de la Room
021    * @param pAllRooms L'arrayList des Rooms ou l'on peux se teleporter
022    */
023   public TransporterRoom(final String pDescription, final String pImage, final int pTypeRoom, final ArrayList<Room> pAllRooms)
024   {
025       super(pDescription, pImage, pTypeRoom);
026       aRoomRandomizer = new RoomRandomizer(pAllRooms);
027   }
028   
029   /**
030    * Retourne une Room al√©atoirement parmis toutes celles disponibles.
031    * @param pDirection La direction dans laquelle on va (ne sers a rien)
032    * @return Une Room aleatoire.
033    */
034   @Override
035   public Room getExit(final String pDirection)
036   {
037       return aRoomRandomizer.getRandomRoom();
038   }
039   
040   /**
041    * Retourne aRoomRandomizer.
042    * @return Le RoomRandomizer
043    */
044   public RoomRandomizer getRoomRandomizer()
045   {
046       return aRoomRandomizer;
047   }
048}
