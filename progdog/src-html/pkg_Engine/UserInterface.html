094    /**
095     * Print out some text into the text area.
096     * @param pText La String a afficher.
097     */
098    public void print( final String pText )
099    {
100        this.aLog.append(" " + pText );
101        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
102    } // print(.)
103
104    /**
105     * Print out some text into the text area, followed by a line break.
106     * @param pText La String a afficher.
107     */
108    public void println( final String pText )
109    {
110        this.print(" " + pText + "\n" );
111    } // println(.)

113    /**
114     * Show an image file in the interface.
115     * @param pImageName La string avec le chemin vers l'image.
116     */
117    public void showImage( final String pImageName )
118    {
119        URL vImageURL = this.getClass().getClassLoader().getResource( pImageName );
120        if ( vImageURL == null )
121            System.out.println( "image not found" );
122        else {
123            ImageIcon vIcon = new ImageIcon( vImageURL );
124            this.aImage.setIcon( vIcon );
125            this.aMyFrame.pack();
126        }
127    } // showImage(.)
128
129    /**
130     * Enable or disable input in the input field.
131     */
132    public void enable( final boolean pOnOff )
133    {
134        this.aEntryField.setEditable( pOnOff );
135        if ( ! pOnOff )
136            this.aEntryField.getCaret().setBlinkRate( 0 );
137        for(JButton vButton : this.aAllButtons)
138           vButton.setEnabled( pOnOff );
139    } // enable(.)

392    /**
393     * Actionlistener interface for entry textfield.
394     * @param pE An ActionEvent
395     */
396    public void actionPerformed( final ActionEvent pE ) 
397    {
398        if(pE.getSource().getClass() == aBoutonWest.getClass())
399        {
400            this.aEngine.interpretCommand( pE.getActionCommand());
401        }
402        else 
403        {   this.processCommand();}
404    } // actionPerformed(.)
405
406    /**
407     * A command has been entered. Read the command and do whatever is 
408     * necessary to process it.
409     */
410    private void processCommand()
411    {
412        String vInput = this.aEntryField.getText();
413        this.aEntryField.setText( "" );
414
415        this.aEngine.interpretCommand( vInput );
416    } // processCommand()
