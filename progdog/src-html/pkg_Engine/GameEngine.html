095    /**
096     * Avec une commande donnee, l'execute.
097     * si cette commande finis le jeu, retourne true, sinon retourne false.
098     * @param pCommandLine le string correspondant a la commande.
099     */
100    public void interpretCommand(String pCommandLine) 
101    {
102        this.aGui.println(pCommandLine);
103        Command vCommand = this.aParser.getCommand(pCommandLine);
104        if(vCommand == null) {
105            this.aGui.println("I don't know what you mean...");
106        } 
107        else {
108            vCommand.execute(this);
109        }
110    }

112    /**
113     * Certaines Rooms ayant des actions a effectuer
114     * (baisser la defence, baisser les PVs, afficher les statistiques du Player, lancer un combat, ...)
115     * Ces actions sont faites ici.
116     */
117    public void RoomAction()
118    {
119        int vTypeRoom = this.aPlayer.getRoomType();
120        if(this.aPlayer.getNbRoomChickenized() == 5)
121        {
122            this.aPlayer.attribuerStats();
123            this.aPlayer.reequip();
124            this.aPlayer.resetNbRoomChickenized();
125            this.aPlayer.setChickenized(false);
126            this.aGui.println("You recovered from the chickenizing spell.");
127        }
128        if(aMap.getMovingNPC()!=null && aMap.getMovingNPC().getCurrentRoom().getType()==20)
129        {
130            if(aMap.getMovingNPC().getCurrentRoom().getNPC("Chicken")!=null)
131            {
132                int aNbRand = (int)Math.floor(Math.random()*21);
133                if(aNbRand>= 15)
134                {
135                    this.aMap.getMovingNPC().getCurrentRoom().getNPC("Chicken").changePV(12);
136                    this.aMap.getMovingNPC().getCurrentRoom().getNPC("Chicken").changeDef(12);
137                    this.aMap.getMovingNPC().getCurrentRoom().getNPC("Chicken").changeAgi(12);
138                    this.aMap.getMovingNPC().getCurrentRoom().getNPC("Chicken").changeAtt(12);
139                    this.aMap.getMovingNPC().getCurrentRoom().getNPC("Chicken").setImage("Images/dovhaPoulet.png");
140                    this.aMap.getMovingNPC().getCurrentRoom().getNPC("Chicken").setNom("Dovah-Chicken");
141                    this.aMap.getMovingNPC().getCurrentRoom().removeMovingNPC();
142                    this.aGui.println("You hear a frightful roar somewhere in the dungeon." );
143                }
144                else
145                    this.aGui.println("burp" );
146            }
147        }
148        switch(vTypeRoom)
149        {
150            case 2 : 
151                this.aGui.println("Suddenly, a wizard appears before you.");
152                int aNbRand = (int)Math.floor(Math.random()*4+1);
153                if (aNbRand == 1)
154                {
155                   this.aGui.println("It's a black wizard, you're now cursed.");
156                   this.aPlayer.changeDef(-2);
157                }
158                else if(aNbRand == 2)
159                {
160                   this.aGui.println("It's a pyromancer, he throws a fireball at you.");
161                   this.aPlayer.changePV(-5);
162                   this.zeroPV();
163                }
164                else if(aNbRand == 3)
165                {
166                   this.aGui.println("It's a light wizard, you're now blessed.");
167                   this.aPlayer.changeDef(+2);
168                }
169                else 
170                {
171                   this.aGui.println("It's a necromancer, the dead bodies that where laying around attack you.");
172                   String[] vTabDial = {"Kak kakak kak."};
173                   this.demarrerCombat( new NPC("Skeleton", "Images/skeleton.jpg", vTabDial, "drakeïde", "assassin") );
174                   this.aGui.setFlee(false);
175                }
176                this.aGui.println("");
177                    break;
178            case 3 :
179                this.aGui.println(this.aPlayer.stats());
180                    break;
181            case 4 :
182                 if(aPlayer.getNPC("Adventurer")!=null)
183                    if(aPlayer.getNPCMessage("Adventurer").equals("Death !"))
184                        this.demarrerCombat(this.aPlayer.getNPC("Adventurer"));   
185                    break;     
186            case 12 : 
187                if(aPlayer.getNPC("Horde of zombies")!=null)
188                    this.demarrerCombat(this.aPlayer.getNPC("Horde of zombies")); 
189                    break;
190            case 13 : 
191                double AgiPlayer = Math.floor(Math.random()*21);
192                double AgiFleche = Math.floor(Math.random()*21);
193                if(this.aPlayer.getAgi() <= 0) 
194                    this.death("You took an arrow in the knee and died on the move.");
195                else if(this.aPlayer.getAgi() + AgiPlayer <= 15 + AgiFleche)
196                {
197                    this.aGui.println("You took an arrow in the back.\n");
198                    this.aPlayer.changePV(-2);
199                    this.aPlayer.changeAgi(-1);
200                }
201                else
202                {
203                    this.aGui.println("You dodged the arrow that were thrown at your back.\n");
204                    this.aPlayer.changeAgi(-1);
205                }
206                    break;
207            case 14 : 
208                if(!aTest)
209                {
210                    this.aPlayer.attribuerStats();
211                    this.aPlayer.reequip();
212                    this.aGui.println("After drinking at the fountain, you feel the energy regaining your body.\n");
213                }
214                    break;
215            case 15 : 
216                AgiPlayer = Math.floor(Math.random()*21);
217                AgiFleche = Math.floor(Math.random()*21);
218                if(this.aPlayer.getAgi() <= 0) 
219                    this.death("You fell into a pit full of spikes and died on the spot.");
220                else if(this.aPlayer.getAgi() + AgiPlayer <= 10 + AgiFleche)
221                {
222                    this.aGui.println("You fell into a pit full of spikes but grabbed the edge at the last second,\n  you just suffer from a flesh wound.\n");
223                    this.aPlayer.changePV(-4);
224                    this.aPlayer.changeAgi(-1);
225                }
226                else
227                {
228                    this.aGui.println("Just before falling into a pit full of spikes, you stopped your move.\n");
229                    this.aPlayer.changeAgi(-1);
230                }
231                    break;
232            case 16 : 
233                if(aPlayer.getNPC("Giant toad")!=null)
234                    this.demarrerCombat(this.aPlayer.getNPC("Giant toad")); 
235                    break;
236            case 17 : 
237                if(aPlayer.getNPC("Fatal Scythe")!=null)
238                {
239                    this.demarrerCombat(this.aPlayer.getNPC("Fatal Scythe")); 
240                    this.aGui.setFlee(false);
241                }
242                    break;
243            case 18 : 
244                if(aPlayer.getNPC("Mounted chicken")!=null)
245                {
246                    this.demarrerCombat(this.aPlayer.getNPC("Mounted chicken")); 
247                }
248                    break;
249            case 19 : 
250                if(!this.aPlayer.chickenized() && !aTest)
251                {
252                    this.aPlayer.setChickenized(true);
253                    this.aPlayer.setPV(2);
254                    this.aPlayer.setAgi(2);
255                    this.aPlayer.setDef(2);
256                    this.aPlayer.setAtt(2);
257                    this.aGui.println("You have been transformed into a chicken by a magic rune.\n");
258                }
259                    break;
260            case 20 : 
261                if(aPlayer.getNPC("Chicken")!=null)
262                {
263                    this.demarrerCombat(this.aPlayer.getNPC("Chicken")); 
264                }
265                else if(aPlayer.getNPC("Dovah-Chicken")!=null)
266                {
267                    this.demarrerCombat(this.aPlayer.getNPC("Dovah-Chicken")); 
268                }
269                    break;
270        }
271    }

273    /**
274     * Lancer le combat contre le dragon (MovingCharacter)
275     */
276    public void combatDragon()
277    {
278        if(this.aPlayer.isMovingCharacter())
279        {
280            if(!aCombat)
281            {
282                this.aGui.combatInterface();
283                this.setCombat(true);
284            }
285            this.aGui.showImage(aPlayer.getMovingCharacter().getImage());
286        }
287    }
288    
289    /**
290     * Lancer le combat contre tout autre ennemi que le dragon.
291     * @param pNPC Le NPC contre qui le Player se bat
292     */
293    public void demarrerCombat(final NPC pNPC)
294    {
295        this.aEnnemi = pNPC;
296        this.aGui.combatInterface();
297        this.setCombat(true);
298        this.aGui.showImage(pNPC.getImage());
299    }
